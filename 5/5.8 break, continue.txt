```
#include <iostream>

using namespace std;

void breakOrReturn()
{
	while (true)
	{
		char ch;
		cin >> ch;

		if (ch == 'b')
			break;

		// return은 아예 함수 자체를 나가버림
		if (ch == 'r')
			return;
		// return 하면 이 부분 실행 X
		cout << "Hello" << endl;
	}
}

int main() 
{
	breakOrReturn();
	return 0;
}
```


do~while과 continue

```
#include <iostream>

using namespace std;

int main() 
{
	int count(0);
	do
	{
		if (count == 5)
			continue;

		cout << count << endl;
	} while (++count < 10);		// 이걸 이렇게 써야 하는 이유가 있다
		// while문 안에 증감을 넣는 방법
	return 0;
}
```


```
#include <iostream>

using namespace std;

int main() 
{
	int count(0);
	do
	{
		if (count == 5)
			continue;

		cout << count << endl;
		count++;
		// 만약 이렇게 넣으면 continue 이후 count++ 부분이 무시가 되어서 무한루프에 빠진다
	} while (count < 10);		

	return 0;
}
```
