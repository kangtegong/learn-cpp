경우에 따라서는 do while문이 꼭 필요한 경우가 있다.

```
#include <iostream>

using namespace std;

int main() 
{
	int selection; // must be declared outside do/while loop
	do
	{
		cout << "1. add" << endl;
		cout << "2. sub" << endl;
		cout << "3. mult" << endl;
		cout << "4. div" << endl;
		cin >> selection;
	} while (selection <= 0 || selection >= 5);

	// do{ } while();에서 세미콜론 잊지말기

	cout << "You selected" << selection << endl;

	return 0;
}
```
